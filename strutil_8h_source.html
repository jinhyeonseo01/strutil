371 static inline std::vector<std::string> split(const std::string & str, const std::string & delim)
372 {
373 size_t pos_start = 0, pos_end, delim_len = delim.length();
374 std::string token;
375 std::vector<std::string> tokens;
376
377 while ((pos_end = str.find(delim, pos_start)) != std::string::npos)
378 {
379 token = str.substr(pos_start, pos_end - pos_start);
380 pos_start = pos_end + delim_len;
381 tokens.push_back(token);
382 }
383
384 tokens.push_back(str.substr(pos_start));
385 return tokens;
